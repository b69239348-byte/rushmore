  ```python
  """Download NBA team logo PNGs from the NBA CDN for local use in generate_card.py."""

  import time
  import urllib.request
  from pathlib import Path

  LOGO_DIR = Path(__file__).parent.parent / "assets" / "team_logos"

  TEAM_IDS = {
      "ATL": 1610612737, "BOS": 1610612738, "BKN": 1610612751,
      "CHA": 1610612766, "CHI": 1610612741, "CLE": 1610612739,
      "DAL": 1610612742, "DEN": 1610612743, "DET": 1610612765,
      "GSW": 1610612744, "HOU": 1610612745, "IND": 1610612754,
      "LAC": 1610612746, "LAL": 1610612747, "MEM": 1610612763,
      "MIA": 1610612748, "MIL": 1610612749, "MIN": 1610612750,
      "NOP": 1610612740, "NYK": 1610612752, "OKC": 1610612760,
      "ORL": 1610612753, "PHI": 1610612755, "PHX": 1610612756,
      "POR": 1610612757, "SAC": 1610612758, "SAS": 1610612759,
      "TOR": 1610612761, "UTA": 1610612762, "WAS": 1610612764,
  }


  def download_logos():
      LOGO_DIR.mkdir(parents=True, exist_ok=True)
      ok, skipped, failed = 0, 0, []
      for abbr, team_id in TEAM_IDS.items():
          path = LOGO_DIR / f"{abbr}.png"
          if path.exists():
              print(f"  {abbr}: skip (exists)")
              skipped += 1
              continue
          url = f"https://cdn.nba.com/logos/nba/{team_id}/primary/L/logo.png"
          try:
              req = urllib.request.Request(url, headers={"User-Agent": "Mozilla/5.0"})
              with urllib.request.urlopen(req, timeout=10) as resp:
                  path.write_bytes(resp.read())
              print(f"  {abbr}: ok")
              ok += 1
              time.sleep(0.3)
          except Exception as e:
              print(f"  {abbr}: FAILED — {e}")
              failed.append(abbr)
      print(f"\nFertig: {ok} geladen, {skipped} übersprungen, {len(failed)} fehlgeschlagen")
      if failed:
          print(f"Fehlgeschlagen: {failed}")


  if __name__ == "__main__":
      print(f"Lade 30 Team-Logos nach {LOGO_DIR}\n")
      download_logos()
  ```

  export const TEAM_NBA_IDS: Record<string, number> = {
    ATL: 1610612737, BOS: 1610612738, BKN: 1610612751,
    CHA: 1610612766, CHI: 1610612741, CLE: 1610612739,
    DAL: 1610612742, DEN: 1610612743, DET: 1610612765,
    GSW: 1610612744, HOU: 1610612745, IND: 1610612754,
    LAC: 1610612746, LAL: 1610612747, MEM: 1610612763,
    MIA: 1610612748, MIL: 1610612749, MIN: 1610612750,
    NOP: 1610612740, NYK: 1610612752, OKC: 1610612760,
    ORL: 1610612753, PHI: 1610612755, PHX: 1610612756,
    POR: 1610612757, SAC: 1610612758, SAS: 1610612759,
    TOR: 1610612761, UTA: 1610612762, WAS: 1610612764,
  };

  /** NBA CDN SVG logo URL for a team abbreviation. Returns null if unknown. */
  export function getTeamLogoUrl(abbr: string | null | undefined): string | null {
    if (!abbr) return null;
    const id = TEAM_NBA_IDS[abbr.toUpperCase()];
    if (!id) return null;
    return `https://cdn.nba.com/logos/nba/${id}/global/L/logo.svg`;
  }

  /** Primary team abbreviation for a player (single team or first in array). */
  export function getPlayerTeam(player: {
    team?: string | null;
    teams?: string[];
  }): string | null {
    return player.team || player.teams?.[0] || null;
  }
  ```

  def _load_team_logo(team_abbr: str, size: int) -> Optional[Image.Image]:
      """Load team logo PNG, resize, and reduce opacity for watermark use."""
      if not team_abbr:
          return None
      path = TEAM_LOGO_DIR / f"{team_abbr.upper()}.png"
      if not path.exists():
          return None
      img = Image.open(path).convert("RGBA")
      img = img.resize((size, size), Image.LANCZOS)
      r, g, b, a = img.split()
      # Reduce to ~7% opacity
      a = a.point(lambda x: int(x * 0.07))
      return Image.merge("RGBA", (r, g, b, a))
  ```
